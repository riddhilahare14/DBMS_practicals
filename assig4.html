üìã Problem Statement

üìñ Library Book Return System

Create a procedure to automate book returns with fine calculation based on days overdue, updating book status from 'I' (Issued) to 'R' (Returned).

üí∞ Fine Calculation Rules

Days 0-15: No fine

Days 16-30: ‚Çπ5 per day

Days > 30: ‚Çπ75 + ‚Çπ50 per day beyond 30

Example (35 days): ‚Çπ75 + (5 √ó ‚Çπ50) = ‚Çπ325

Step 2 
DELIMITER $$

CREATE PROCEDURE ReturnBook (
    IN p_rollno INT,
    IN p_book_name VARCHAR(50)
)
BEGIN
    DECLARE v_date_of_issue DATE;
    DECLARE v_status CHAR(1);
    DECLARE v_days INT;
    DECLARE v_fine_amt DECIMAL(10, 2) DEFAULT 0;
    
    DECLARE exit handler FOR NOT FOUND 
    BEGIN
        SELECT 'No matching record found.' AS Message;
    END;
    
    SELECT date_of_issue, status 
    INTO v_date_of_issue, v_status
    FROM Borrower
    WHERE roll_no = p_rollno AND book_name = p_book_name;
    
    IF v_status = 'I' THEN
        SET v_days = DATEDIFF(CURDATE(), v_date_of_issue);
        
        IF v_days > 30 THEN
            SET v_fine_amt = (15 * 5) + ((v_days - 30) * 50);
        ELSEIF v_days > 15 THEN
            SET v_fine_amt = (v_days - 15) * 5;
        ELSE
            SET v_fine_amt = 0;
        END IF;
        
        UPDATE Borrower
        SET status = 'R'
        WHERE roll_no = p_rollno AND book_name = p_book_name;
        
        IF v_fine_amt > 0 THEN
            INSERT INTO Fine (roll_no, fine_date, fine_amt)
            VALUES (p_rollno, CURDATE(), v_fine_amt);
        END IF;
        
        SELECT 'Book returned successfully.' AS Message, v_fine_amt AS Fine;
    ELSE
        SELECT 'Book already returned.' AS Message;
    END IF;
END $$

DELIMITER ;
